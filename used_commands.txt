# 2 - Iniciar o serviÃ§o com o comando: 
mysqld_safe --skip-grant-tables & 

# 3 - Acessar o banco 
mysql -u root 

Executar os comando abaixo

mysql> use mysql;
mysql> flush privileges; 
mysql> update user set password=PASSWORD("NOVASENHA") where User='root';
mysql> flush privileges; 
mysql> quit;

ou ainda
